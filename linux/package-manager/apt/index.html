    apt-cache search apache2 | grep -Ei "(php|mysql|security2) "   

    apt-cache search apache2 |grep -o -Ei "^[^[:space:]]+(php|mysql|security2)[[:space:]]"

    apt-cache search apache2|awk '/^[^[:space:]]+(php|mysql|security2)[[:space:]]/ {print $1}'
