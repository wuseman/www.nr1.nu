    Yet Another Rename (bash function)¶

    rename(){ 
    txtToReplace=${1} ; 
    replacementTxt=${2} ; 
    shift 2 ; 
    files=${@} ; 
    for file in $files ; do 
        mv ${file} ${file/${txtToReplace}/${replacementTxt}} ; 
    done 
}
