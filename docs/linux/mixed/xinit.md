# xinit

##### start a nested awesome session

   xinit  $(which awesome)  /usr/bin/Xephyr :5 -once -fullscreen -ac -keybd "ephyr,,,xkbmodel=pc105,xkblayout=it,xkbrules=evdev,xkboption="

##### from the console, start a second X server

   xinit   :1

##### from the console, start a second X server

   xinit   :1
